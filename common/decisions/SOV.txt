### CIRCASSIA - Integrity 5 trigger | No stability req.
	SOV_mission_circassia_secession = {

		icon = generic_civil_support

		allowed = {
			original_tag = SOV
		}
		
		available = {
				OR = {
				has_idea = RUS_integrity_6
				has_completed_focus = SOV_Total_Isolationism
				country_exists = CIR
				}
		}
		
		visible = {
			tag = SOV
		}

		days_mission_timeout = 80
		#days_remove = 1

		activation = {
				OR = {
				has_idea = RUS_integrity_5
				has_idea = RUS_integrity_4
				has_idea = RUS_integrity_3
				has_idea = RUS_integrity_2
				has_idea = RUS_integrity_1
				}
			NOT = { country_exists = CIR }
		}

		fixed_random_seed = no

		fire_only_once = no

		is_good = no

		timeout_effect = {
			if = {
				# 1. We go independent
				limit = {
				OR = {
				has_idea = RUS_integrity_4
				has_idea = RUS_integrity_3
				has_idea = RUS_integrity_2
				has_idea = RUS_integrity_1
				}
				}
				country_event = {
					id = secession.1
				}
			}
			else_if = {
				# 2. We go conditional
				limit = {
					has_idea = RUS_integrity_5
				}
				country_event = {
					id = secession.2
				}
			}
		}
	}
